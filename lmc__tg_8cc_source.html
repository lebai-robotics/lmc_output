153  }

159  tg_runtime_data->time_cost = interp_state.running_time;

161  }
162  for (unsigned int i = 0; i < interp_state.jnt_pose.rows(); ++i) {
163  joint_cmd[i].pos_cmd = interp_state.jnt_pose(i);
164  joint_cmd[i].vel_cmd = interp_state.jnt_vel(i);
